"use client";

import React from "react";
import { PageNavigation } from "@/components/page-navigation";
import { RadioPlayer } from "@/components/radio-player";
import { useQuery } from "@tanstack/react-query";
import { RadioEpisode } from "@/app/api/radio/route";
import { ExternalLink } from "lucide-react";
import { Button } from "@/components/ui/button";
import Link from "next/link";
import Image from "next/image";

export default function RadioPage() {
  const { data: episodes, isLoading } = useQuery<RadioEpisode[]>({
    queryKey: ["radio-shows"],
    queryFn: async () => {
      const response = await fetch("/api/radio");
      return response.json();
    }
  });

  return (
    <div className="w-full">
      <div className="w-full max-w-screen-lg mx-auto -mt-6 px-4 md:px-8 relative z-10 mb-8 md:mb-16">
        <PageNavigation />

        {isLoading ? (
          <div className="space-y-6">
            <div className="bg-white dark:bg-zinc-800 rounded-2xl border-2 border-white dark:border-zinc-700 p-6 animate-pulse">
              <div className="h-8 bg-zinc-200 dark:bg-zinc-700 rounded w-1/3 mb-4"></div>
              <div className="h-4 bg-zinc-200 dark:bg-zinc-700 rounded w-full mb-2"></div>
              <div className="h-4 bg-zinc-200 dark:bg-zinc-700 rounded w-2/3"></div>
            </div>
          </div>
        ) : (
          <div className="space-y-6">
            {episodes?.map((episode) => (
              <div key={episode.id}>
                {episode.status === "available" ? (
                  <RadioPlayer
                    src={episode.audioUrl}
                    title={episode.title}
                    episode={episode.episode}
                    date={episode.date}
                    hosts={episode.hosts}
                    guests={episode.guests}
                  />
                ) : (
                  <div className="bg-white dark:bg-zinc-800 rounded-2xl border-2 border-white dark:border-zinc-700 shadow-lg overflow-hidden">
                    <div className="p-4 md:p-6">
                      {/* Header - Compact */}
                      <div className="flex items-center justify-between mb-4">
                        <div className="flex items-center gap-2">
                          <span className="text-xs font-semibold px-2 py-1 bg-blue-500 text-white rounded-md">
                            {episode.episode}
                          </span>
                          <h2 className="text-lg md:text-xl font-bold text-foreground">{episode.title}</h2>
                        </div>
                        <span className="text-xs text-muted-foreground">{episode.date}</span>
                      </div>

                      {/* Hosts and Guests */}
                      <div className="flex items-center gap-6 mb-4">
                        {/* Hosts */}
                        <div className="flex items-center gap-2">
                          <p className="text-xs font-semibold text-muted-foreground">HOSTED BY</p>
                          <div className="flex -space-x-2">
                            {episode.hosts.map((host, index) => (
                              <div
                                key={index}
                                className="group relative"
                              >
                                <div className="relative w-8 h-8 rounded-full overflow-hidden border-2 border-white dark:border-zinc-800 hover:z-10 transition-transform hover:scale-110">
                                  <Image
                                    src={host.image}
                                    alt={host.name}
                                    fill
                                    className="object-cover"
                                  />
                                </div>
                                {/* Tooltip */}
                                <div className="absolute bottom-full left-1/2 -translate-x-1/2 mb-2 px-2 py-1 bg-zinc-900 text-white text-xs rounded opacity-0 group-hover:opacity-100 transition-opacity pointer-events-none whitespace-nowrap z-20">
                                  {host.name}
                                </div>
                              </div>
                            ))}
                          </div>
                        </div>

                        {/* Guests */}
                        {episode.guests && episode.guests.length > 0 && (
                          <div className="flex items-center gap-2">
                            <p className="text-xs font-semibold text-muted-foreground">GUEST</p>
                            <div className="flex -space-x-2">
                              {episode.guests.map((guest, index) => (
                                <div
                                  key={index}
                                  className="group relative"
                                >
                                  <div className="relative w-8 h-8 rounded-full overflow-hidden border-2 border-white dark:border-zinc-800 hover:z-10 transition-transform hover:scale-110">
                                    <Image
                                      src={guest.image}
                                      alt={guest.name}
                                      fill
                                      className="object-cover"
                                    />
                                  </div>
                                  {/* Tooltip */}
                                  <div className="absolute bottom-full left-1/2 -translate-x-1/2 mb-2 px-2 py-1 bg-zinc-900 text-white text-xs rounded opacity-0 group-hover:opacity-100 transition-opacity pointer-events-none whitespace-nowrap z-20">
                                    {guest.name}
                                  </div>
                                </div>
                              ))}
                            </div>
                          </div>
                        )}
                      </div>

                      {episode.twitterUrl && (
                        <Link href={episode.twitterUrl} target="_blank" rel="noopener noreferrer">
                          <Button variant="outline" className="w-full sm:w-auto">
                            <ExternalLink className="h-4 w-4 mr-2" />
                            View Announcement on X
                          </Button>
                        </Link>
                      )}
                    </div>
                  </div>
                )}
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
